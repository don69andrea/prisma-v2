### Task 2: Crypto dashboard smoke test

Create `frontend/tests/e2e/crypto-dashboard.spec.ts`:

```typescript
import { test, expect } from '@playwright/test';

test.describe('Crypto Dashboard', () => {
  test('overview page loads with at least 1 signal', async ({ page }) => {
    await page.goto('/crypto');
    await page.waitForSelector('table', { timeout: 15000 });
    // Check disclaimer visible
    await expect(page.getByText('kein Anlagerat')).toBeVisible();
    // Check at least one signal badge (BUY, HOLD, or SELL)
    const badges = page.locator('text=/^(BUY|HOLD|SELL)$/');
    await expect(badges.first()).toBeVisible({ timeout: 10000 });
  });

  test('coin detail page loads for BTC', async ({ page }) => {
    await page.goto('/crypto/btc');
    await expect(page.getByText('Übersicht & Erklärung')).toBeVisible({ timeout: 10000 });
    await expect(page.getByText('Schicht 1 — WAS')).toBeVisible({ timeout: 10000 });
  });

  test('navigation has Krypto-Signale link', async ({ page }) => {
    await page.goto('/');
    await expect(page.getByRole('link', { name: /krypto/i })).toBeVisible();
  });
});
```
