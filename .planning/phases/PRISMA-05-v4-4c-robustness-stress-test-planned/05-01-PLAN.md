    FUNCTION build_signals(close: pd.Series, ma_window: int = 100) -> pd.Series:
    - Compute sma_val = sma(close, window=ma_window)
    - Compute rsi_14 = rsi(close, window=14)
    - Compute _, _, macd_hist = macd(close)
    - ma_s = (close > sma_val).astype(float).shift(1).fillna(0.0)  # shift(1) BEFORE consensus — D-09
    - rsi_s = (rsi_14 > 50).astype(float).shift(1).fillna(0.0)
    - macd_s = (macd_hist > 0).astype(float).shift(1).fillna(0.0)
    - signals_df = pd.DataFrame({"ma_signal": ma_s, "rsi_signal": rsi_s, "macd_signal": macd_s})
    - consensus = consensus_vote(signals_df)
    - rv = close.pct_change().rolling(21).std() * (252 ** 0.5)  # rolling(21) approximation (D-10)
    - rolling_vol = rv.shift(1).fillna(0.60)
    - size_factors = rolling_vol.apply(lambda v: vol_target_size(v, target_vol=0.60, cap=1.5))
    - return consensus.astype(float) * size_factors

    FUNCTION _download(coin: str, start: str, end: str | None = None) -> pd.Series | None:
    - raw = yf.download(coin, start=start, end=end, progress=False, auto_adjust=True)
    - if len(raw) == 0: return None
    - close = raw["Close"].squeeze()
    - close.index = pd.DatetimeIndex(close.index).tz_localize("UTC")
    - return close

    FUNCTION _bah_metrics(close: pd.Series) -> tuple[float, float]:
    - Buy-and-Hold = run_walkforward_with_details with all-ones signals and costs=0.0001
    - prices_df = pd.DataFrame({"close": close})
    - bah_signals = pd.Series(1.0, index=close.index)
    - details = run_walkforward_with_details(prices_df, bah_signals, costs=0.0001)
    - return _sharpe(details["net_returns"]), _calmar(details["net_returns"])

    FUNCTION run_cost_sensitivity(coins=_COST_COINS, cost_levels=_COST_LEVELS, ma_window=_DEFAULT_MA) -> list[CostResult | dict]:
    - For each coin: download from _DATA_START; skip if None or len<_MIN_ROWS (return insufficient dict)
    - bah_sharpe, bah_calmar = _bah_metrics(close)
    - signals = build_signals(close, ma_window=ma_window)
    - prices_df = pd.DataFrame({"close": close})
    - For each cost in cost_levels: run_walkforward_with_details(prices_df, signals, costs=cost)
    - Append CostResult with beats_exposure_matched from details["beats_exposure_matched"]

    FUNCTION run_regime_splits(coins=_CRYPTO_UNIVERSE, regimes=_REGIMES, costs=_DEFAULT_COST, ma_window=_DEFAULT_MA) -> list[RegimeResult | dict]:
    - For each coin, for each regime:
    - Download from _DATA_START to regime["end"] — NOT just the regime window (anti-pattern per RESEARCH.md)
    - If len(close) < _MIN_ROWS: append {"status":"insufficient","coin":coin,"regime":regime["name"],"rows":len,"reason":"not listed"}
    - If coin is MATIC-USD and regime["end"] > "2025-03-24": use end="2025-03-24" and add note="MATIC-USD ends 2025-03-24 (renamed POL)"
    - build_signals and run_walkforward_with_details normally
    - Slice OOS returns: mask = (net.index >= regime["start"]) & (net.index <= regime["end"])
    - regime_net = net[mask]; regime_base = baseline[mask]
    - If len(regime_net) < 21: append {"status":"no_oos_in_regime","coin":coin,"regime":regime["name"],"oos_rows":len}
    - Else: append RegimeResult with _sharpe/_calmar/_max_drawdown on sliced series

    FUNCTION run_universe(ma_window=_DEFAULT_MA, costs=_DEFAULT_COST) -> list[UniverseResult | dict]:
    - For each coin in _CRYPTO_UNIVERSE:
    - Download from _DATA_START; if None return download_failed dict; if len<_MIN_ROWS return insufficient dict
    - For MATIC-USD: add note="MATIC-USD ends 2025-03-24 (renamed POL)"
    - build_signals + run_walkforward_with_details
    - Append UniverseResult

    FUNCTION run_parameter_stability(coins=_PARAM_COINS, ma_windows=_MA_WINDOWS, costs=_DEFAULT_COST) -> list[ParamResult | dict]:
    - For each coin: download once from _DATA_START (download outside window loop to avoid repeated downloads)
    - For each window in ma_windows:
    - build_signals with ma_window=window
    - run_walkforward_with_details with costs=costs
    - is_default = window == _DEFAULT_MA  # marks the anchor point (D-01)
    - Append ParamResult
